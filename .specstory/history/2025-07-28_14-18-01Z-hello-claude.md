    85→def migrate_config(settings: Dict[str, Any]) -> Dict[str, Any]:
    86→    """Migrate configuration from older versions to the current version."""
    87→    # Ensure settings structure exists
    88→    if "settings" not in settings:
    89→        settings["settings"] = {}
    90→
    91→    # Check for version in both meta and _meta sections
    92→    has_version = False
    93→    if "meta" in settings["settings"] and "version" in settings["settings"]["meta"]:
    94→        has_version = True
    95→    elif "_meta" in settings["settings"] and "version" in settings["settings"]["_meta"]:
    96→        has_version = True
    97→
    98→    if not has_version:
    99→        # This is an old config without versioning
   100→        console.print("[yellow]Migrating configuration to the latest version...[/]")
   101→
   102→        # Add metadata section with current version
   103→        settings["settings"]["meta"] = {
   104→            "version": CONFIG_VERSION,
   105→            "last_updated": time.time(),
   106→        }
   107→
   108→        # Add new sections that didn't exist in older versions
   109→        if "module" not in settings["settings"]:
   110→            # Initialize empty module settings
   111→            settings["settings"]["module"] = {"history": []}
   112→
   113→        if "credentials" not in settings:
   114→            settings["credentials"] = {}
   115→
   116→    # Migrate chat history from old location to new location if needed
   117→    if "chat" in settings and "history" in settings["chat"]:
   118→        # Ensure module section exists
   119→        if "module" not in settings["settings"]:
   120→            settings["settings"]["module"] = {}
   121→
   122→        # Ensure history section exists in module
   123→        if "history" not in settings["settings"]["module"]:
   124→            settings["settings"]["module"]["history"] = []
   125→
   126→        # Copy history from old location to new location
   127→        settings["settings"]["module"]["history"] = settings["chat"]["history"]
   128→
   129→    # Handle future migrations based on version numbers
   130→    # version = settings["settings"]["meta"].get("version", 0)
   131→
   132→    # Example of future migration:
   133→    # if version < 2:
   134→    #     # Migrate from version 1 to 2
   135→    #     settings["new_section"] = {"new_key": "new_value"}
   136→    #     settings["_meta"]["version"] = 2

   138→    return settings

   141→def load_settings() -> Dict[str, Any]:
   142→    """Load settings from the config file, or create default settings if the file doesn't exist."""
   143→    config_file = get_config_file()
   144→
   145→    if not config_file.exists():
   146→        # Create default config
   147→        settings = load_default_settings()
   148→        settings["settings"]["meta"] = {
   149→            "last_updated": time.time(),
   150→            "version": CONFIG_VERSION,
   151→        }
   152→        save_settings(settings)
   153→        return settings
   154→
   155→    try:
   156→        with open(config_file, "r") as f:
   157→            settings = json.load(f)
   158→
   159→        # Migrate configuration if needed
   160→        settings = migrate_config(settings)
   161→
   162→        # Ensure all required sections exist
   163→        defaults = load_default_settings()
   164→        for section, section_defaults in defaults.items():
   165→            if section not in settings:
   166→                settings[section] = section_defaults
   167→
   168→        return settings
   169→    except json.JSONDecodeError:
   170→        console.print("[bold red]Error:[/] Config file is corrupted.")
   171→
   172→        # Try to restore from backup
   173→        backup = restore_from_latest_backup()
   174→        if backup:
   175→            console.print("[green]Restored configuration from backup.[/]")
   176→            return backup
   177→
   178→        # If no backup, reset to defaults
   179→        console.print("[yellow]Resetting to default configuration.[/]")
   180→        settings = load_default_settings()
   181→        if "settings" not in settings:
   182→            settings["settings"] = {}
   183→        if "meta" not in settings["settings"]:
   184→            settings["settings"]["meta"] = {}
   185→        settings["settings"]["meta"]["last_updated"] = time.time()
   186→        save_settings(settings)
   187→        return settings
   188→    except Exception as e:
   189→        console.print(f"[bold red]Error:[/] Failed to load configuration: {str(e)}")
   190→        return load_default_settings()

   210→def restore_from_latest_backup() -> Optional[Dict[str, Any]]:
   211→    """Attempt to restore configuration from the most recent backup."""
   212→    backup_dir = get_backup_dir()
   213→    if not backup_dir.exists():
   214→        return None
   215→
   216→    # Find the most recent backup file
   217→    backup_files = list(backup_dir.glob("config.json.*.bak"))
   218→    if not backup_files:
   219→        return None
   220→
   221→    # Sort by modification time (newest first)
   222→    backup_files.sort(key=lambda p: p.stat().st_mtime, reverse=True)
   223→
   224→    # Try to load from the most recent backup
   225→    try:
   226→        with open(backup_files[0], "r") as f:
   227→            return cast(Dict[str, Any], json.load(f))
   228→    except Exception:

   293→def get_setting(key: str) -> Any:
   294→    """Get a setting value by its key (supports dot notation for nested settings)."""
   295→    # Check for test environment variable override first (highest priority)
   296→    test_env_key = f"SKOGAI_TEST_{key.upper().replace('.', '_')}"
   297→    env_value = os.getenv(test_env_key)
   298→
   299→    if env_value is None:
   300→        # Check for regular environment variable override
   301→        env_key = f"SKOGAI_{key.upper().replace('.', '_')}"
   302→        env_value = os.getenv(env_key)
   303→    if env_value is not None:
   304→        # Try to parse as different types
   305→        if env_value.lower() in ("true", "yes", "1"):
   306→            return True
   307→        elif env_value.lower() in ("false", "no", "0"):
   308→            return False
   309→        elif env_value.lower() in ("null", "none"):
   310→            return None
   311→        try:
   312→            # Try as int
   313→            return int(env_value)
   314→        except ValueError:
   315→            try:
   316→                # Try as float
   317→                return float(env_value)
   318→            except ValueError:
   319→                # Return as string
   320→                return env_value
   321→
   322→    settings = load_settings()
   323→
   324→    # Handle SkogAI $ syntax - access definitions in the $ section
   325→    if key == "$":
   326→        # Return the entire $ section
   327→        return settings.get("$", {})
   328→    elif key.startswith("$."):
   329→        # Access nested keys within $ section
   330→        dollar_key = key[2:]  # Remove "$." prefix
   331→        dollar_section = settings.get("$", {})
   332→
   333→        # Handle nested access like $.claude.hello
   334→        if "." in dollar_key:
   335→            current = dollar_section
   336→            for part in dollar_key.split("."):
   337→                if not isinstance(current, dict) or part not in current:
   338→                    return None
   339→                current = current[part]
   340→            return current
   341→        else:
   342→            return dollar_section.get(dollar_key)
   343→
   344→    # Handle credentials separately
   345→    if key.startswith("credentials."):
   346→        sensitive_settings = load_sensitive_settings()
   347→        credential_key = key.split(".", 1)[1]
   348→        return sensitive_settings.get(credential_key)
   349→
   350→    if "." in key:
   351→        # Handle nested keys - read directly from root settings (matches set_setting)
   352→        parts = key.split(".")
   353→        current = settings
   354→
   355→        for part in parts:
   356→            if part not in current:
   357→                # Fallback: try reading from nested settings.* structure for backward compatibility
   358→                if parts[0] not in ["settings", "credentials"]:
   359→                    fallback_current = settings.get("settings", {})
   360→                    for fallback_part in parts:
   361→                        if fallback_part not in fallback_current:
   362→                            return None
   363→                        fallback_current = fallback_current[fallback_part]
   364→                    return fallback_current
   365→                return None
   366→            current = current[part]
   367→        return current
   368→
   369→    # For single keys, read directly from root settings (matches set_setting)
   370→    value = settings.get(key)
   371→    if value is not None:
   372→        return value
   373→
   374→    # Fallback: try reading from nested settings.* structure for backward compatibility
   375→    return settings.get("settings", {}).get(key)
   376→

   378→def set_setting(key: str, value: Any) -> bool:
   379→    """Set a setting value by its key (supports dot notation for nested settings).
   380→
   381→    Returns:
   382→        bool: True if successful, False otherwise
   383→    """
   384→    settings = load_settings()
   385→
   386→    # Special handling for credentials
   387→    is_credential = key.startswith("credentials.")
   388→
   389→    if is_credential:
   390→        # Load sensitive settings separately
   391→        sensitive_settings = load_sensitive_settings()
   392→        credential_key = key.split(".", 1)[1]
   393→        sensitive_settings[credential_key] = value
   394→
   395→        # Save to the sensitive file
   396→        try:
   397→            sensitive_file = get_sensitive_config_file()
   398→            with open(sensitive_file, "w") as f:
   399→                json.dump(sensitive_settings, f, indent=2)
   400→
   401→            # Also update the main settings object for consistency
   402→            if "credentials" not in settings:
   403→                settings["credentials"] = {}
   404→            settings["credentials"][credential_key] = value
   405→        except Exception as e:
   406→            console.print(f"[bold red]Error:[/] Failed to save credential: {str(e)}")
   407→            return False
   408→    elif "." in key:
   409→        # Handle nested keys - store directly in root settings
   410→        parts = key.split(".")
   411→        current = settings
   412→
   413→        for i, part in enumerate(parts[:-1]):
   414→            if part not in current:
   415→                current[part] = {}
   416→            current = current[part]
   417→
   418→        # Set the value at the final level
   419→        current[parts[-1]] = value
   420→    else:
   421→        # For single keys, set directly in root settings
   422→        settings[key] = value
   423→
   424→    return save_settings(settings)

def create_backup(config_file: Path) -> Optional[Path]:
    """Create a backup of the configuration file."""
    if not config_file.exists():

    timestamp = time.strftime("%Y%m%d_%H%M%S")

        return None

def restore_from_latest_backup() -> Optional[Dict[str, Any]]:
    """Attempt to restore configuration from the most recent backup."""
    backup_dir = get_backup_dir()

    # Find the most recent backup file
    backup_files = list(backup_dir.glob("config.json.*.bak"))
    if not backup_files:
        return None

    # Sort by modification time (newest first)
    backup_files.sort(key=lambda p: p.stat().st_mtime, reverse=True)

    # Try to load from the most recent backup
    try:
        with open(backup_files[0], "r") as f:
            return cast(Dict[str, Any], json.load(f))

@config_app.command("export-env", help="export environment variables")
def export_env(
    namespace: Optional[str] = typer.Option(
        None, "--namespace", "-n", help="Filter by namespace (e.g., 'claude' for claude.env.* variables)"
    ),
    output: Optional[Path] = typer.Option(
        None, "--output", "-o", help="Write to file instead of stdout"
    ),
):
    """Export environment variables as bash export statements."""
    settings = load_settings()

    def extract_env_variables(
        data: Dict[str, Any], prefix: str = ""
    ) -> List[tuple[str, Any]]:
        """Extract all *.env.* keys from configuration."""
        env_vars = []
        for key, value in data.items():
            full_key = f"{prefix}{key}" if prefix else key
            if isinstance(value, dict):
                env_vars.extend(extract_env_variables(value, f"{full_key}."))
            elif ".env." in full_key:
                env_vars.append((full_key, value))
        return env_vars

    # Get all environment variables
    all_env_vars = extract_env_variables(settings)

    # Filter by namespace if specified
    if namespace:
        namespace_prefix = f"{namespace}.env."
        filtered_vars = [
            (key, value) for key, value in all_env_vars 
            if key.startswith(namespace_prefix)
        ]
        all_env_vars = filtered_vars

    # Filter out null/None values (these represent unset variables)
    valid_vars = [(key, value) for key, value in all_env_vars if value is not None and value != "null"]

    if not valid_vars:
        if namespace:
            console.print(f"[yellow]No environment variables found for namespace '{namespace}'.[/]", err=True)
        else:
            console.print("[yellow]No environment variables found.[/]", err=True)
        return

    # Generate export statements
    export_lines = []
    for key, value in sorted(valid_vars):
        # Extract the actual environment variable name from the config key
        # e.g., "claude.env.MYENV" -> "MYENV"
        env_name = key.split(".env.")[-1]
        
        # Properly escape the value for bash
        if isinstance(value, str):
            # Escape double quotes and backslashes in string values
            escaped_value = value.replace('\\', '\\\\').replace('"', '\\"')
            export_line = f'export {env_name}="{escaped_value}"'
        else:
            # For non-string values, convert to string and quote
            export_line = f'export {env_name}="{str(value)}"'
            
        export_lines.append(export_line)

    # Output to file or stdout
    if output:
        try:
            with open(output, 'w') as f:
                for line in export_lines:
                    f.write(line + '\n')
            console.print(f"[green]Environment variables exported to {output}[/]")
        except Exception as e:
            console.print(f"[bold red]Error:[/] Failed to write to file: {str(e)}")
            return
    else:
        # Print to stdout
        for line in export_lines:
            print(line)
